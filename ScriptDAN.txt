local function findGameDependencies()
    local attempts = 0
    while not (framework and scrollHandler and network) and attempts < 10 do
        for _, obj in next, getgc(true) do
            if type(obj) == 'table' then
                if rawget(obj, 'GameUI') and type(rawget(obj, 'GameUI')) == 'table' then
                    framework = obj;
                elseif type(rawget(obj, 'Server')) == 'table' then
                    network = obj;
                end
            end
        end

        for _, module in next, getloadedmodules() do
            if module.Name == 'ScrollHandler' then
                scrollHandler = module;
                break;
            end
        end

        if not (framework and scrollHandler and network) then
            task.wait(0.1)
            attempts = attempts + 1
        end
    end

    if not (framework and scrollHandler and network) then
        error(string.format('Failed to load critical game dependencies after %d attempts. Details: Framework:%s, ScrollHandler:%s, Network:%s',
            attempts, type(framework), typeof(scrollHandler), type(network)))
    end
end

local ARROW_MAPPINGS = {
    [Enum.KeyCode.D] = 1,
    [Enum.KeyCode.F] = 2,
    [Enum.KeyCode.J] = 3,
    [Enum.KeyCode.K] = 4,
}

local REVERSE_ARROW_MAPPINGS = {}
for k, v in pairs(ARROW_MAPPINGS) do REVERSE_ARROW_MAPPINGS[v] = k end

local function pressKey(keyCode)
    if Autoplayer.PressMode == 'firesignal' then
        network.Server:Input(network.Server, "ArrowPressed", keyCode)
    elseif Autoplayer.PressMode == 'virtual input' then
        userInputService:SimulateKeyPress(keyCode, true)
    end
end

local function releaseKey(keyCode)
    if Autoplayer.PressMode == 'firesignal' then
        network.Server:Input(network.Server, "ArrowReleased", keyCode)
    elseif Autoplayer.PressMode == 'virtual input' then
        userInputService:SimulateKeyPress(keyCode, false)
    end
end

local function getAdjustedDelay(isHeld)
    local baseDelay = isHeld and Autoplayer.HeldDelay or Autoplayer.ReleaseDelay
    if Autoplayer.DelayMode == 'Aleatório' then
        local minDelay = isHeld and Autoplayer.HeldDelayMin or Autoplayer.NoteDelayMin
        local maxDelay = isHeld and Autoplayer.HeldDelayMax or Autoplayer.NoteDelayMax
        baseDelay = math.random(minDelay, maxDelay)
    end
    local humanError = math.random(-5, 5)
    return math.max(0, baseDelay + humanError) / 1000
end

local function handleNote(noteData)
    if not Autoplayer.Enabled then return end
    if math.random(1, 100) <= Autoplayer.HumanErrorChance then return end

    local currentTime = tick()
    if currentTime - Autoplayer.LastHitTime < Autoplayer.MinHitInterval then return end

    local arrowIndex = noteData.Arrow
    local keyCode = REVERSE_ARROW_MAPPINGS[arrowIndex]

    if keyCode then
        local delayToPress = 0
        local releaseDuration = getAdjustedDelay(noteData.IsHeld)

        task.delay(delayToPress, function()
            pressKey(keyCode)
            task.delay(releaseDuration, function()
                releaseKey(keyCode)
            end)
            Autoplayer.LastHitTime = tick()
        end)
    end
end

-- Save Manager (adapted for VenyxUI - relies on Venyx internal option system)
local SaveManager = {}
SaveManager.FileName = 'dan_player_config.json'

function SaveManager:SaveConfig(configName)
    local data = {
        version = 3,
        options = {}
    }
    for key, element in pairs(Main.Elements) do -- Iterate VenyxUI elements
        if element.Type and element.Value ~= nil then -- Only elements with value
            data.options[key] = element.Value
        end
    end
    local success, encoded = pcall(httpService.JSONEncode, httpService, data)
    if not success then return false, 'Falha ao codificar dados: '.. tostring(encoded) end
    writefile('funky_friday_autoplayer/'.. (configName or SaveManager.FileName), encoded)
    return true
end

function SaveManager:LoadConfig(configName)
    local filePath = 'funky_friday_autoplayer/'.. (configName or SaveManager.FileName)
    if not isfile(filePath) then return false, 'Arquivo de configuração não encontrado.' end

    local success, decoded = pcall(httpService.JSONDecode, httpService, readfile(filePath))
    if not success then return false, 'Erro ao decodificar configuração: '.. tostring(decoded) end
    if decoded.version ~= 3 then return false, 'Versão de configuração incompatível! (Esperado 3, Encontrado '.. tostring(decoded.version).. ')' end

    for key, value in pairs(decoded.options) do
        local element = Main.Elements[key]
        if element and element.SetValue then -- Check if element exists and has SetValue method
            element:SetValue(value)
        end
    end
    return true
end

function SaveManager.RefreshConfigList(dropdown)
    local list = listfiles('funky_friday_autoplayer')
    local out = {}
    for i = 1, #list do
        local file = list[i]
        if file:sub(-5) == '.json' then
            local filename = file:match("([^/\\\]+)%.json$")
            if filename and filename ~= SaveManager.FileName:match("([^/\\\]+)%.json$") then -- Exclude default
                table.insert(out, filename)
            end
        end
    end
    dropdown:SetValues(out)
    if #out > 0 then dropdown:SetValue(out[1]) end
end

    return math.max(0, baseDelay + humanError) / 1000
end

local function handleNote(noteData)
    if not Autoplayer.Enabled then return end
    if math.random(1, 100) <= Autoplayer.HumanErrorChance then return end -- Simula erro humano

    local currentTime = tick()
    if currentTime - Autoplayer.LastHitTime < Autoplayer.MinHitInterval then return end

    local arrowIndex = noteData.Arrow
    local keyCode = REVERSE_ARROW_MAPPINGS[arrowIndex]

    if keyCode then
        local delayToPress = 0
        local releaseDuration = getAdjustedDelay(noteData.IsHeld)

        task.delay(delayToPress, function()
            pressKey(keyCode)
            task.delay(releaseDuration, function()
                releaseKey(keyCode)
            end)
            Autoplayer.LastHitTime = tick()
        end)
    end
end

function SaveManager:Save(name)
    local fullPath = 'funky_friday_autoplayer/configs/'.. name.. '.json'
    local data = { version = 3, objects = {} }

    local uiElements = {}
    for idx, toggle in pairs(Toggles or {}) do if toggle.Type and not self.Ignore[idx] then table.insert(uiElements, {idx = idx, obj = toggle}) end end
    for idx, option in pairs(Options or {}) do if option.Type and not self.Ignore[idx] then table.insert(uiElements, {idx = idx, obj = option}) end end

    for _, element in ipairs(uiElements) do
        if self.Parser[element.obj.Type] then
            table.insert(data.objects, self.Parser[element.obj.Type].Save(element.idx, element.obj))
        end
    end

    local success, encoded = pcall(httpService.JSONEncode, httpService, data)
    if not success then return false, 'Falha ao codificar dados: '.. tostring(encoded) end
    writefile(fullPath, encoded)
    return true
end

function SaveManager:Load(name)
    local file = 'funky_friday_autoplayer/configs/'.. name.. '.json'
    if not isfile(file) then return false, 'Arquivo de configuração não encontrado.' end

    local success, decoded = pcall(httpService.JSONDecode, httpService, readfile(file))
    if not success then return false, 'Erro ao decodificar configuração: '.. tostring(decoded) end
    if decoded.version ~= 3 then return false, 'Versão de configuração incompatível! (Esperado 3, Encontrado '.. tostring(decoded.version).. ')' end

    for _, optionData in next, decoded.objects do
        if self.Parser[optionData.type] then
            local targetObject = Options[optionData.idx] or Toggles[optionData.idx]
            if targetObject then
                 self.Parser[optionData.type].Load(optionData.idx, optionData)
            end
        end
    end
    return true
end

function SaveManager.Refresh()
    local list = listfiles('funky_friday_autoplayer/configs')
    local out = {}
    for i = 1, #list do
        local file = list[i]
        if file:sub(-5) == '.json' then
            local filename = file:match("([^/\\\]+)%.json$")
            if filename then table.insert(out, filename) end
        end
    end
    Options.ConfigList.Values = out;
    Options.ConfigList:SetValues()
    Options.ConfigList:Display()
    if #out > 0 then Options.ConfigList:Set(out[1]) end
end

function SaveManager.Check()
    local list = listfiles('funky_friday_autoplayer/configs')
    for _, file in next, list do
        if isfolder(file) then continue end
        local data = readfile(file)
        local success, decoded = pcall(httpService.JSONDecode, httpService, data)
        if success and type(decoded) == 'table' and decoded.version ~= 3 then
            pcall(delfile, file)
        end
    end
end
