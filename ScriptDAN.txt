local Players = game:GetService("Players")
local TweenService = game:GetService("TweenService")

local character = player.Character or player.CharacterAdded:Wait()

-- =================== BOTÃO FLUTUANTE MAIS BONITO ===================

ScreenGui.Name = "DAN_PS99_v7_0"
ScreenGui.ResetOnSpawn = false
ScreenGui.Parent = player:WaitForChild("PlayerGui")

local FloatButton = Instance.new("TextButton")
FloatButton.Size = UDim2.new(0, 95, 0, 95)
FloatButton.Position = UDim2.new(0.84, 0, 0.62, 0)
FloatButton.BackgroundColor3 = Color3.fromRGB(0, 90, 255)
FloatButton.Text = "🐾\nPS99\nDAN"
FloatButton.TextScaled = true
FloatButton.Font = Enum.Font.GothamBlack

Instance.new("UICorner", FloatButton).CornerRadius = UDim.new(1, 0)

-- Sombra forte

end)

print("✅ ANTI-AFK ATIVADO AUTOMATICAMENTE")

-- =================== AO CLICAR NO BOTÃO - MUITOS HUBS ===================
FloatButton.MouseButton1Click:Connect(function()
    print("🔥 DAN v7.0 - INICIANDO CARREGAMENTO DE MÚLTIPLOS HUBS...")
    
    -- Hub 1: Redz Hub (principal)
    local success1 = pcall(function()
        loadstring(game:HttpGet("https://raw.githubusercontent.com/REDzHUB/PetSimulator99/main/redz9999.lua"))()
        print("✅ Redz Hub carregado com sucesso!")
    end)
    
    if not success1 then
        print("⚠️ Redz falhou, tentando Zap Hub...")
        pcall(function()
            loadstring(game:HttpGet('https://zaphub.xyz/Exec'))()
            print("✅ Zap Hub carregado!")
        end)
    end
    
    -- Hub 2: Bubblescripts (bom para mobile)
    task.wait(1)
    pcall(function()
        loadstring(game:HttpGet("https://raw.githubusercontent.com/bubblescripts/scripts/refs/heads/main/PS99/psgo"))()
        print("✅ Bubblescripts carregado!")
    end)
    
    -- Hub 3: SlamminPig (auto farm forte)
    task.wait(1.5)
    pcall(function()
        loadstring(game:HttpGet('https://raw.githubusercontent.com/SlamminPig/6FootScripts/main/Scripts/PetSimulator99.lua'))()
        print("✅ SlamminPig carregado!")
    end)
    
    -- Hub 4: HugeGames (se quiser key, mas tenta)
    task.wait(1)
    pcall(function()
        loadstring(game:HttpGet("https://HugeGames.io/ps99"))()
        print("✅ HugeGames tentado!")
    end)
    
    -- Notificação grande e detalhada
    local NotifFrame = Instance.new("Frame")
    NotifFrame.Size = UDim2.new(0, 380, 0, 140)
    NotifFrame.Position = UDim2.new(0.5, -190, 0.35, 0)
    NotifFrame.BackgroundColor3 = Color3.fromRGB(20, 120, 80)
    NotifFrame.Parent = ScreenGui
    Instance.new("UICorner", NotifFrame).CornerRadius = UDim.new(0, 18)
    
    local NotifText = Instance.new("TextLabel")
    NotifText.Size = UDim2.new(1, -20, 1, -20)
    NotifText.Position = UDim2.new(0, 10, 0, 10)
    NotifText.BackgroundTransparency = 1
    NotifText.Text = "✅ DAN v7.0\nMÚLTIPLOS HUBS CARREGADOS!\n\nUse qualquer menu que apareceu:\n• Auto Farm\n• Auto Hatch\n• Fusion / Upgrade\n• ESP\n• Rebirth\n• Infinite Speed\n\nBoa farm, chefe!"
    NotifText.TextColor3 = Color3.new(1,1,1)
    NotifText.TextScaled = true
    NotifText.Font = Enum.Font.GothamBold
    NotifText.TextWrapped = true
    NotifText.Parent = NotifFrame
    
    task.delay(8, function()
        if NotifFrame then NotifFrame:Destroy() end
    end)
    
    print("=====================================")
    print("DAN v7.0 FINALIZADO - MUITOS HUBS TENTADOS")
    print("Toque nos menus que apareceram e ligue as funções desejadas.")
    print("Se precisar de mais fallbacks ou ajustes, só falar!")
end)

    end
end)
