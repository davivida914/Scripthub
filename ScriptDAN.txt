local Players = game:GetService("Players")
local TweenService = game:GetService("TweenService")
local UserInputService = game:GetService("UserInputService")

local player = Players.LocalPlayer

-- =================== BOTÃO FLUTUANTE MELHORADO ===================
local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "DAN_PS99_v6_3"
ScreenGui.ResetOnSpawn = false
ScreenGui.Parent = player:WaitForChild("PlayerGui")

local FloatButton = Instance.new("TextButton")
FloatButton.Size = UDim2.new(0, 85, 0, 85)
FloatButton.Position = UDim2.new(0.86, 0, 0.68, 0)
FloatButton.BackgroundColor3 = Color3.fromRGB(0, 110, 255)
FloatButton.Text = "🐾\nPS99"
FloatButton.TextScaled = true
FloatButton.Font = Enum.Font.GothamBlack
FloatButton.TextColor3 = Color3.new(1,1,1)
FloatButton.Parent = ScreenGui

Instance.new("UICorner", FloatButton).CornerRadius = UDim.new(1, 0)

-- Sombra no botão
local Shadow = Instance.new("Frame")
Shadow.Size = UDim2.new(1, 14, 1, 14)
Shadow.Position = UDim2.new(0, -7, 0, -7)
Shadow.BackgroundColor3 = Color3.fromRGB(0,0,0)
Shadow.BackgroundTransparency = 0.65
Shadow.ZIndex = -1
Shadow.Parent = FloatButton
Instance.new("UICorner", Shadow).CornerRadius = UDim.new(1,0)

-- =================== MENU PRINCIPAL ===================
local MainFrame = Instance.new("Frame")
MainFrame.Size = UDim2.new(0, 430, 0, 540)
MainFrame.Position = UDim2.new(0.5, -215, 0.22, 0)
MainFrame.BackgroundColor3 = Color3.fromRGB(18, 18, 27)
MainFrame.Visible = false
MainFrame.Parent = ScreenGui
Instance.new("UICorner", MainFrame).CornerRadius = UDim.new(0, 20)

-- Arrastar com dedo (melhorado)
local dragging = false
MainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
    end
end)

MainFrame.InputChanged:Connect(function(input)
    if dragging and (input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseMovement) then
        MainFrame.Position = UDim2.new(0, input.Position.X - 215, 0, input.Position.Y - 270)
    end
end)

MainFrame.InputEnded:Connect(function() dragging = false end)

-- Título melhorado
local Title = Instance.new("TextLabel")
Title.Size = UDim2.new(1, 0, 0, 70)
Title.BackgroundColor3 = Color3.fromRGB(30, 30, 80)
Title.Text = "🐾 DAN PS99 v6.3\nMELHORADO - FUNCIONANDO 2026"
Title.TextColor3 = Color3.new(1,1,1)
Title.TextScaled = true
Title.Font = Enum.Font.GothamBlack
Title.Parent = MainFrame
Instance.new("UICorner", Title).CornerRadius = UDim.new(0, 20)

-- Botão fechar
local CloseBtn = Instance.new("TextButton")
CloseBtn.Size = UDim2.new(0, 55, 0, 55)
CloseBtn.Position = UDim2.new(1, -68, 0, 8)
CloseBtn.BackgroundColor3 = Color3.fromRGB(200, 40, 40)
CloseBtn.Text = "✕"
CloseBtn.TextColor3 = Color3.new(1,1,1)
CloseBtn.TextScaled = true
CloseBtn.Font = Enum.Font.GothamBold
CloseBtn.Parent = Title
Instance.new("UICorner", CloseBtn).CornerRadius = UDim.new(1,0)

CloseBtn.MouseButton1Click:Connect(function()
    MainFrame.Visible = false
end)

-- =================== CARREGAMENTO DOS HUBS ===================
FloatButton.MouseButton1Click:Connect(function()
    MainFrame.Visible = not MainFrame.Visible
    
    if MainFrame.Visible then
        print("🔥 DAN v6.3 - Tentando carregar hub real...")
        
        -- Tenta Redz Hub primeiro (mais citado como working)
        local success, err = pcall(function()
            loadstring(game:HttpGet("https://raw.githubusercontent.com/REDzHUB/PetSimulator99/main/redz9999.lua"))()
        end)
        
        if not success then
            print("⚠️ Redz Hub falhou, tentando fallback...")
            -- Fallback: outro hub popular
            pcall(function()
                loadstring(game:HttpGet("https://raw.githubusercontent.com/bubblescripts/scripts/refs/heads/main/PS99/psgo"))()
            end)
        end
        
        -- Notificação bonita
        local Notif = Instance.new("TextLabel")
        Notif.Size = UDim2.new(0, 360, 0, 70)
        Notif.Position = UDim2.new(0.5, -180, 0.65, 0)
        Notif.BackgroundColor3 = Color3.fromRGB(0, 160, 80)
        Notif.Text = "✅ Hub carregado!\nUse as opções do menu que apareceu.\nBoa farm!"
        Notif.TextColor3 = Color3.new(1,1,1)
        Notif.TextScaled = true
        Notif.Font = Enum.Font.GothamBold
        Notif.Parent = MainFrame
        Instance.new("UICorner", Notif).CornerRadius = UDim.new(0, 16)
        
        task.delay(6, function()
            if Notif and Notif.Parent then Notif:Destroy() end
        end)
    end
end)

print("✅ DAN SCRIPT v6.3 MELHORADO CARREGADO!")
print("Toque no botão flutuante 🐾 para abrir e ativar o hub.")
print("Redz Hub + fallback ativados. Teste e me diga o que rolou!")
