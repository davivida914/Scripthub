    valueLabel.Text = tostring(config.Default or 0).. (config.Suffix or "")
    valueLabel.Font = Enum.Font.SourceSansSemibold
    valueLabel.TextSize = 14
    valueLabel.Parent = frame

    local sliderFrame = Instance.new("Frame")
    sliderFrame.Size = UDim2.new(1, -10, 0, 10)
    sliderFrame.Position = UDim2.new(0, 5, 0.5, 0)
    sliderFrame.BackgroundColor3 = Themes[currentTheme].SecondaryColor
    sliderFrame.BorderColor3 = Themes[currentTheme].BorderColor
    sliderFrame.BorderSizePixel = 1
    sliderFrame.Parent = frame

    local fillFrame = Instance.new("Frame")
    fillFrame.Size = UDim2.new(0, 0, 1, 0)
    fillFrame.BackgroundColor3 = Themes[currentTheme].AccentColor
    fillFrame.Parent = sliderFrame

    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 3)
    corner.Parent = sliderFrame

    local fillCorner = Instance.new("UICorner")
    fillCorner.CornerRadius = UDim.new(0, 3)
    fillCorner.Parent = fillFrame

    local value = config.Default or 0
    local min = config.Min or 0
    local max = config.Max or 100
    local rounding = config.Rounding or 0

    local function updateSlider(mouseX)
        local relativeX = math.clamp(mouseX - sliderFrame.AbsolutePosition.X, 0, sliderFrame.AbsoluteSize.X)
        local percentage = relativeX / sliderFrame.AbsoluteSize.X
        value = min + (max - min) * percentage
        value = math.floor(value * (10 ^ rounding) + 0.5) / (10 ^ rounding) -- Rounding

        fillFrame.Size = UDim2.new(percentage, 0, 1, 0)
        valueLabel.Text = tostring(value).. (config.Suffix or "")
        if config.OnChanged then config.OnChanged(value) end
        UI_ELEMENTS[config.Name].Value = value
    end

    sliderFrame.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            updateSlider(input.Position.X)
            local connection = game:GetService("UserInputService").InputChanged:Connect(function(input2)
                if input2.UserInputType == Enum.UserInputType.MouseMovement or input2.UserInputType == Enum.UserInputType.Touch then
                    updateSlider(input2.Position.X)
                end
            end)
            local connection2 = game:GetService("UserInputService").InputEnded:Connect(function(input2)
                if input2.UserInputType == Enum.UserInputType.MouseButton1 or input2.UserInputType == Enum.UserInputType.Touch then
                    connection:Disconnect()
                    connection2:Disconnect()
                end
            end)
        end
    end)

    local element = {
        Type = "Slider",
        Value = value,
        OnChanged = function(callback) config.OnChanged = callback end,
        SetValue = function(newValue)
            value = math.clamp(newValue, min, max)
            value = math.floor(value * (10 ^ rounding) + 0.5) / (10 ^ rounding)
            local percentage = (value - min) / (max - min)
            fillFrame.Size = UDim2.new(percentage, 0, 1, 0)
            valueLabel.Text = tostring(value).. (config.Suffix or "")
            UI_ELEMENTS[config.Name].Value = value
        end
    }
    UI_ELEMENTS[config.Name] = element

    function element:ApplyTheme(theme)
        frame.BackgroundColor3 = theme.MainColor
        textLabel.BackgroundColor3 = theme.MainColor
        textLabel.TextColor3 = theme.TextColor
        valueLabel.BackgroundColor3 = theme.MainColor
        valueLabel.TextColor3 = theme.TextColor
        sliderFrame.BackgroundColor3 = theme.SecondaryColor
        sliderFrame.BorderColor3 = theme.BorderColor
        fillFrame.BackgroundColor3 = theme.AccentColor
    end
    table.insert(UI_COMPONENTS, element)
    return element
end

    dropdownButton.TextXAlignment = Enum.TextXAlignment.Left

    dropdownButton.Text = config.Default or (config.Values and config.Values[1]) or "Select..."
    dropdownButton.Font = Enum.Font.SourceSansSemibold
    dropdownButton.TextSize = 14
    dropdownButton.Parent = frame

    local arrowLabel = Instance.new("TextLabel")
    arrowLabel.Size = UDim2.new(0, 20, 1, 0)
    arrowLabel.Position = UDim2.new(1, -25, 0, 0)
    arrowLabel.BackgroundColor3 = Themes[currentTheme].SecondaryColor
    arrowLabel.TextColor3 = Themes[currentTheme].TextColor
    arrowLabel.Text = "▼"
    arrowLabel.Font = Enum.Font.SourceSansSemibold
    arrowLabel.TextSize = 14
    arrowLabel.Parent = dropdownButton

    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 3)
    corner.Parent = dropdownButton

    local dropdownFrame = Instance.new("Frame")
    dropdownFrame.Size = UDim2.new(1, 0, 0, 0) -- Starts collapsed
    dropdownFrame.Position = UDim2.new(0, 0, 1, 5)
    dropdownFrame.BackgroundColor3 = Themes[currentTheme].SecondaryColor
    dropdownFrame.BorderColor3 = Themes[currentTheme].BorderColor
    dropdownFrame.BorderSizePixel = 1
    dropdownFrame.Parent = frame
    dropdownFrame.Visible = false

    local listLayout = Instance.new("UIListLayout")
    listLayout.FillDirection = Enum.FillDirection.Vertical
    listLayout.Padding = UDim.new(0, 2)
    listLayout.Parent = dropdownFrame

    local value = config.Default or (config.Values and config.Values[1])
    local isExpanded = false

    local function populateDropdown()
        for _, child in pairs(dropdownFrame:GetChildren()) do
            if child:IsA("TextButton") then child:Destroy() end
        end
        for _, val in pairs(config.Values or {}) do
            local optionButton = createButton(dropdownFrame, val, function()
                value = val
                dropdownButton.Text = val
                dropdownFrame.Visible = false
                isExpanded = false
                if config.OnChanged then config.OnChanged(value) end
                UI_ELEMENTS[config.Name].Value = value
            end)
            optionButton.TextXAlignment = Enum.TextXAlignment.Left
            optionButton.BackgroundColor3 = Themes[currentTheme].SecondaryColor
            optionButton.TextColor3 = Themes[currentTheme].TextColor
            optionButton.Size = UDim2.new(1, 0, 0, 25)
            optionButton.Parent = dropdownFrame
        end
        dropdownFrame.Size = UDim2.new(1, 0, 0, (#config.Values or 0) * 27 + 4)
    end
    populateDropdown()

    dropdownButton.MouseButton1Click:Connect(function()
        isExpanded = not isExpanded
        dropdownFrame.Visible = isExpanded
    end)

    local element = {
        Type = "Dropdown",
        Value = value,
        OnChanged = function(callback) config.OnChanged = callback end,
        SetValue = function(newValue)
            if table.find(config.Values, newValue) then
                value = newValue
                dropdownButton.Text = newValue
                UI_ELEMENTS[config.Name].Value = value
            else
                warn("Value not found in dropdown options:", newValue)
            end
        end,
        SetValues = function(newValues)
            config.Values = newValues
            populateDropdown()
        end
    }
    UI_ELEMENTS[config.Name] = element

    function element:ApplyTheme(theme)
        frame.BackgroundColor3 = theme.MainColor
        dropdownButton.BackgroundColor3 = theme.SecondaryColor
        dropdownButton.BorderColor3 = theme.BorderColor
        dropdownButton.TextColor3 = theme.TextColor
        arrowLabel.BackgroundColor3 = theme.SecondaryColor
        arrowLabel.TextColor3 = theme.TextColor
        dropdownFrame.BackgroundColor3 = theme.SecondaryColor
        dropdownFrame.BorderColor3 = theme.BorderColor
        for _, child in pairs(dropdownFrame:GetChildren()) do
            if child:IsA("TextButton") and child.ApplyTheme then
                child:ApplyTheme(theme)
            end
        end
    end
    table.insert(UI_COMPONENTS, element)
    return element
end

local function findGameDependencies()
    local attempts = 0
    while not (framework and scrollHandler and network) and attempts < 10 do
        for _, obj in next, getgc(true) do
            if type(obj) == 'table' then
                if rawget(obj, 'GameUI') and type(rawget(obj, 'GameUI')) == 'table' then
                    framework = obj;
                elseif type(rawget(obj, 'Server')) == 'table' then
                    network = obj;
                end
            end
        end

        for _, module in next, getloadedmodules() do
            if module.Name == 'ScrollHandler' then
                scrollHandler = module;
                break;
            end
        end

        if not (framework and scrollHandler and network) then
            task.wait(0.1)
            attempts = attempts + 1
        end
    end

    if not (framework and scrollHandler and network) then
        error(string.format('Failed to load critical game dependencies after %d attempts. Details: Framework:%s, ScrollHandler:%s, Network:%s',
            attempts, type(framework), typeof(scrollHandler), type(network)))
    end
end

findGameDependencies()

    end

    if type(debug.getinfo) ~= 'function' then
        _G.debug.getinfo = function(f)

local ARROW_MAPPINGS = {
    [Enum.KeyCode.D] = 1,
    [Enum.KeyCode.F] = 2,
    [Enum.KeyCode.J] = 3,
    [Enum.KeyCode.K] = 4,
}

local REVERSE_ARROW_MAPPINGS = {}
for k, v in pairs(ARROW_MAPPINGS) do REVERSE_ARROW_MAPPINGS[v] = k end

local function pressKey(keyCode)
    if Autoplayer.PressMode == 'firesignal' then
        network.Server:Input(network.Server, "ArrowPressed", keyCode)
    elseif Autoplayer.PressMode == 'virtual input' then
        userInputService:SimulateKeyPress(keyCode, true)
    end
end

local function releaseKey(keyCode)
    if Autoplayer.PressMode == 'firesignal' then
        network.Server:Input(network.Server, "ArrowReleased", keyCode)
    elseif Autoplayer.PressMode == 'virtual input' then
        userInputService:SimulateKeyPress(keyCode, false)
    end
end

local function getAdjustedDelay(isHeld)
    local baseDelay = isHeld and Autoplayer.HeldDelay or Autoplayer.ReleaseDelay
    if Autoplayer.DelayMode == 'Aleatório' then
        local minDelay = isHeld and Autoplayer.HeldDelayMin or Autoplayer.NoteDelayMin
        local maxDelay = isHeld and Autoplayer.HeldDelayMax or Autoplayer.NoteDelayMax
        baseDelay = math.random(minDelay, maxDelay)
    end
    local humanError = math.random(-5, 5)
    return math.max(0, baseDelay + humanError) / 1000
end

local function handleNote(noteData)
    if not Autoplayer.Enabled then return end
    if math.random(1, 100) <= Autoplayer.HumanErrorChance then return end

    local currentTime = tick()
    if currentTime - Autoplayer.LastHitTime < Autoplayer.MinHitInterval then return end

    local arrowIndex = noteData.Arrow
    local keyCode = REVERSE_ARROW_MAPPINGS[arrowIndex]

    if keyCode then
        local delayToPress = 0
        local releaseDuration = getAdjustedDelay(noteData.IsHeld)

        task.delay(delayToPress, function()
            pressKey(keyCode)
            task.delay(releaseDuration, function()
                releaseKey(keyCode)
            end)
            Autoplayer.LastHitTime = tick()
        end)
    end
end

function SaveManager:SaveConfig(configName)
    local data = {
        version = 3,
        options = {}
    }
    for key, element in pairs(MainUI.Elements) do -- Iterate VenyxUI elements
        if element.Type and element.Value ~= nil then -- Only elements with value
            data.options[key] = element.Value
        end
    end
    local success, encoded = pcall(httpService.JSONEncode, httpService, data)
    if not success then return false, 'Falha ao codificar dados: '.. tostring(encoded) end
    makefolder('funky_friday_autoplayer')
    writefile('funky_friday_autoplayer/'.. configName, encoded)
    return true
end

function SaveManager:LoadConfig(configName)
    local filePath = 'funky_friday_autoplayer/'.. configName
    if not isfile(filePath) then return false, 'Arquivo de configuração não encontrado.' end

    local success, decoded = pcall(httpService.JSONDecode, httpService, readfile(filePath))
    if not success then return false, 'Erro ao decodificar configuração: '.. tostring(decoded) end
    if decoded.version ~= 3 then return false, 'Versão de configuração incompatível! (Esperado 3, Encontrado '.. tostring(decoded.version).. ')' end

    for key, value in pairs(decoded.options) do
        local element = MainUI.Elements[key]
        if element and element.SetValue then -- Check if element exists and has SetValue method
            element:SetValue(value)
        end
    end
    return true
end

            local filename = file:match("([^/\\\]+)$") -- Get full filename with extension
            table.insert(out, filename)
        end
    end
    dropdown:SetValues(out)
end

-- Save Manager (adapted for VenyxUI - relies on Venyx internal option system)
local SaveManager = {}
SaveManager.FileName = 'dan_player_config.json'

function SaveManager:SaveConfig(configName)
    local data = {
        version = 3,
        options = {}
    }
    for key, element in pairs(Main.Elements) do -- Iterate VenyxUI elements
        if element.Type and element.Value ~= nil then -- Only elements with value
            data.options[key] = element.Value
        end
    end
    local success, encoded = pcall(httpService.JSONEncode, httpService, data)
    if not success then return false, 'Falha ao codificar dados: '.. tostring(encoded) end
    writefile('funky_friday_autoplayer/'.. (configName or SaveManager.FileName), encoded)
    return true
end

function SaveManager:LoadConfig(configName)
    local filePath = 'funky_friday_autoplayer/'.. (configName or SaveManager.FileName)
    if not isfile(filePath) then return false, 'Arquivo de configuração não encontrado.' end

    local success, decoded = pcall(httpService.JSONDecode, httpService, readfile(filePath))
    if not success then return false, 'Erro ao decodificar configuração: '.. tostring(decoded) end
    if decoded.version ~= 3 then return false, 'Versão de configuração incompatível! (Esperado 3, Encontrado '.. tostring(decoded.version).. ')' end

    for key, value in pairs(decoded.options) do
        local element = Main.Elements[key]
        if element and element.SetValue then -- Check if element exists and has SetValue method
            element:SetValue(value)
        end
    end
    return true
end

function SaveManager.RefreshConfigList(dropdown)
    local list = listfiles('funky_friday_autoplayer')
    local out = {}
    for i = 1, #list do
        local file = list[i]
        if file:sub(-5) == '.json' then
            local filename = file:match("([^/\\\]+)%.json$")
            if filename and filename ~= SaveManager.FileName:match("([^/\\\]+)%.json$") then -- Exclude default
                table.insert(out, filename)
            end
        end
    end
    dropdown:SetValues(out)
    if #out > 0 then dropdown:SetValue(out[1]) end
end
